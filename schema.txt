drop table user
create table user(
   Uid int AUTO_INCREMENT PRIMARY KEY,
    id varchar(50) ,
    phonenumber varchar(50) ,
    email varchar(50) ,
    password varchar(50) ,
    address1 varchar(50) ,
    address2 varchar(50) ,
    address3 varchar(50) ,
    isPermit int ,
    accountnumber varchar(50) ,
    role varchar(50)
);

drop table company
create table company(
   Cid int AUTO_INCREMENT PRIMARY KEY,
    Uid varchar(50) ,
    registration_number varchar(50) ,
    manager_name varchar(50) ,
    manager_phonenumber varchar(50) ,
    interestproduct varchar(50) ,
    typeofcompany varchar(50) 
);

drop table management
create table management(
   Cid int AUTO_INCREMENT PRIMARY KEY,
    Uid varchar(50) ,
    registration_number varchar(50) ,
    manager_name varchar(50) ,
    manager_phonenumber varchar(50) 
);

drop table star
create table star(
   sid int AUTO_INCREMENT PRIMARY KEY,
    uid varchar(50) ,
    management varchar(50) ,
    favoritem varchar(50) ,
    SNSurl varchar(50) ,
    type int ,
    wage varchar(50) 
);

drop table starbody

    registration_number varchar(50) ,
    manager_name varchar(50) ,

drop table starimage
create table starimage(

drop table item

drop table itemimage

drop table matching

drop table notice
create table notice(
